# allo-maison.ma Phase 1 MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship allo-maison.ma Phase 1 MVP — homepage + 36 service-city pages + 18 emergency pages + WhatsApp booking flow + full SEO.

**Architecture:** Next.js App Router with static data files for MVP (no database needed yet — artisan profiles are manually curated seed data). Programmatic pages via `generateStaticParams()`. Booking flow redirects to WhatsApp with pre-filled message. Schema markup on every page.

**Tech Stack:** Next.js 15 (App Router), Tailwind CSS v4, shadcn/ui, Vercel, GA4, Resend (contact form)

**Reference:** Design spec at `docs/superpowers/specs/2026-04-11-allo-maison-platform-design.md`. Homepage mockup at `.superpowers/brainstorm/26242-1775942179/content/homepage-full.html`.

import type { MetadataRoute } from "next";
import { SERVICES } from "@/lib/data/services";
import { CITIES } from "@/lib/data/cities";

const BASE_URL = process.env.NEXT_PUBLIC_SITE_URL || "https://allo-maison.ma";
const MVP_SERVICES = ["plombier", "electricien", "femme-de-menage", "peintre", "climatisation", "serrurier"];
const MVP_CITIES = ["casablanca", "rabat", "marrakech", "tanger", "fes", "agadir"];
const URGENCE_SLUGS = ["plombier", "electricien", "serrurier"];

export default function sitemap(): MetadataRoute.Sitemap {
  const entries: MetadataRoute.Sitemap = [
    { url: BASE_URL, lastModified: new Date(), changeFrequency: "weekly", priority: 1.0 },
  ];

  // Service x City pages
  for (const s of MVP_SERVICES) {
    for (const c of MVP_CITIES) {
      entries.push({
        url: `${BASE_URL}/${s}-${c}`,
        lastModified: new Date(),
        changeFrequency: "weekly",
        priority: 0.8,
      });
    }
  }

  // Urgence pages
  for (const s of URGENCE_SLUGS) {
    for (const c of MVP_CITIES) {
      entries.push({
        url: `${BASE_URL}/urgence/${s}/${c}`,
        lastModified: new Date(),
        changeFrequency: "weekly",
        priority: 0.9,
      });
    }
  }

  return entries;
}
